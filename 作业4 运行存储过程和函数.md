    DELIMITER $$
    CREATE FUNCTION func_employee_sal (empno INT)  
        RETURNS DOUBLE
    BEGIN
        RETURN (SELECT sal 
            FROM t_employee2 
            WHERE t_employee2.empno = empno);
    END$$
